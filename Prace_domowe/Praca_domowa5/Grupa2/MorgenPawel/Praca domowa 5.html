from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans, DBSCAN, AgglomerativeClustering
from sklearn.metrics import davies_bouldin_score, silhouette_score

sc = StandardScaler()
df_scaled = sc.fit_transform(df)
max_k = 15
results_k = np.ndarray((max_k - 1,6))
for k in range(2,max_k + 1):
    kmeans = KMeans(n_clusters = k)
    agg = AgglomerativeClustering(n_clusters = k, linkage = 'average')
    ward = AgglomerativeClustering(n_clusters = k, linkage = 'ward')
    kmeans_labels = kmeans.fit_predict(df_scaled)
    agg_labels = agg.fit_predict(df_scaled)
    ward_labels = ward.fit_predict(df_scaled)
    kmeans_db = davies_bouldin_score(df_scaled, kmeans_labels)
    kmeans_sil = silhouette_score(df_scaled, kmeans_labels)
    agg_db = davies_bouldin_score(df_scaled, agg_labels)
    agg_sil = silhouette_score(df_scaled, agg_labels)
    ward_db = davies_bouldin_score(df_scaled, ward_labels)
    ward_sil = silhouette_score(df_scaled, ward_labels)
    results_k[k-2] = [kmeans_db, kmeans_sil, agg_db, agg_sil, ward_db, ward_sil]
# results_k
results_pd = pd.DataFrame(data = results_k)
results_pd.columns = ["kmeans_0", "kmeans_1", "agg_avg_0", "agg_avg_1", "agg_ward_0", "agg_ward_1"]
results_pd["id"] = results_pd.index
results_pd["k"] = np.array(range(2, max_k + 1))
cols = ['agg_avg', 'kmeans', 'agg_ward']
df1 = pd.wide_to_long(results_pd, stubnames = cols, i = 'id', j = 'metric', sep = '_').reset_index(level = 'metric').reset_index(drop = True).melt(id_vars = ['metric', 'k'], var_name = 'algoritm')
df1 = df1.assign(metric = np.array(['davies_bouldin', 'silhouette'])[df1.loc[:,'metric']])

Pamitajmy, e w indeksie Daviesa-Bouldina dymy do jak najmniejszej wartoci, a w metryce silhouette - najwikszej. Std widzimy, e w obu metrykach optymalna ilo klastrw to 8. Porwnajmy ten wynik z algorytmem DBSCAN.

epses = np.power(10, np.linspace(-1.5, -0.1))
results_eps = np.ndarray((len(epses), 4))
for i in range(len(epses)):
    eps = epses[i]
    dbscan = DBSCAN(eps)
    dbscan_labels = dbscan.fit_predict(df_scaled)
    k = len(np.unique(dbscan_labels[dbscan_labels != -1]))
    if(k > 1 and (np.sum(dbscan_labels == -1) / len(dbscan_labels) <= 0.2)):
        # Uwaga: kolumny oznaczone jako "-1" nie nale do adnego klastra. 
        # Nie powinno si ich bra pod uwag przy liczeniu metryk.
        # Z drugiej strony, nie chcemy, by byo zbyt duo "-1".
        # Uznajemy, e podziay, przy ktrych wicej ni 20% obserwacji ma "-1", s bezwartociowe.
        dbscan_db = davies_bouldin_score(df_scaled[dbscan_labels != -1, :], dbscan_labels[dbscan_labels != -1])
        dbscan_sil = silhouette_score(df_scaled[dbscan_labels != -1, :], dbscan_labels[dbscan_labels != -1])
    else:
        dbscan_db = 1
        dbscan_sil = 0
    results_eps[i] = [dbscan_db, dbscan_sil, k, eps]
df_eps = pd.DataFrame(results_eps)
df_eps.columns = ["davies_bouldin", "silhouette", "k", "eps"]
df_eps = df_eps.melt(id_vars = ['k', 'eps'], var_name = 'metric')
